# Sentiment Arbitrage v7.3 技術改版方案

## AI 有效性評估、A/B 測試與模組考核系統

**版本**: v7.3
**目標**: 建立可量化的模組貢獻度評估體系，用數據驅動功能取捨
**預計週期**: 4-6 週

---

## 概述

當前 v7.2 系統存在的核心問題：
1. **缺乏歸因機制**：一筆交易盈利，無法判斷是哪個模組的貢獻
2. **AI 模組效果未知**：AINarrativeSystem、SmartExitEngine AI 等模組無驗證數據
3. **無 A/B 測試基礎**：無法對比"使用 vs 不使用"某功能的效果
4. **自動調參缺乏數據支撐**：SelfIterationManager 有調權能力但缺乏歸因數據

本方案分三階段實施：
- **階段一**：數據基礎建設（1-2 週）
- **階段二**：實驗與評估框架（1-2 週）
- **階段三**：自動調權與模組考核（1-2 週）

### 任務 2.1：創建模組績效分析腳本

**文件位置**：`scripts/analyze-module-performance.js`

```javascript
#!/usr/bin/env node
/**
 * 模組績效分析腳本
 *
 * 功能：
 * 1. 按 entry_source 分組計算績效
 * 2. 按 ai_narrative_tier 分組計算績效
 * 3. 計算各模組相對於基準的表現
 * 4. 輸出報告和建議
 */

import Database from 'better-sqlite3';

const db = new Database('./data/sentiment_arb.db');

function analyzeBySource(windowDays = 7) {
    const cutoff = new Date(Date.now() - windowDays * 24 * 60 * 60 * 1000).toISOString();

    // 整體基準
    const baseline = db.prepare(`
        SELECT
            COUNT(*) as total,
            AVG(exit_pnl_percent) as avg_pnl,
            SUM(CASE WHEN exit_pnl_percent >= 50 THEN 1 ELSE 0 END) * 100.0 / COUNT(*) as win_rate
        FROM positions
        WHERE status = 'closed' AND exit_time >= ?
    `).get(cutoff);

    // 按 entry_source 分組
    const bySource = db.prepare(`
        SELECT
            entry_source,
            COUNT(*) as trades,
            AVG(exit_pnl_percent) as avg_pnl,
            SUM(CASE WHEN exit_pnl_percent >= 50 THEN 1 ELSE 0 END) as wins,
            SUM(CASE WHEN exit_pnl_percent >= 50 THEN 1 ELSE 0 END) * 100.0 / COUNT(*) as win_rate,
            MAX(exit_pnl_percent) as best,
            MIN(exit_pnl_percent) as worst
        FROM positions
        WHERE status = 'closed' AND exit_time >= ?
        GROUP BY entry_source
        ORDER BY avg_pnl DESC
    `).all(cutoff);

    console.log(`\n${'═'.repeat(70)}`);
    console.log(`📊 模組績效分析 (過去 ${windowDays} 天)`);
    console.log(`${'═'.repeat(70)}`);
    console.log(`\n基準: ${baseline.total} 筆交易, 胜率 ${baseline.win_rate?.toFixed(1)}%, 平均收益 ${baseline.avg_pnl?.toFixed(1)}%\n`);

    console.log('按入場來源分組:');
    console.log('-'.repeat(70));

    for (const row of bySource) {
        const relPerf = baseline.avg_pnl ?
            ((row.avg_pnl - baseline.avg_pnl) / Math.abs(baseline.avg_pnl) * 100).toFixed(0) : 0;
        const emoji = row.avg_pnl > baseline.avg_pnl ? '✅' : row.avg_pnl < 0 ? '❌' : '⚠️';

        console.log(`${emoji} ${row.entry_source || 'unknown'}`);
        console.log(`   交易: ${row.trades} | 胜率: ${row.win_rate?.toFixed(1)}% | 平均: ${row.avg_pnl?.toFixed(1)}% | 相對: ${relPerf}%`);
        console.log(`   最佳: +${row.best?.toFixed(0)}% | 最差: ${row.worst?.toFixed(0)}%`);
    }

    return { baseline, bySource };
}

function analyzeByNarrativeTier(windowDays = 7) {
    const cutoff = new Date(Date.now() - windowDays * 24 * 60 * 60 * 1000).toISOString();

    const byTier = db.prepare(`
        SELECT
            ai_narrative_tier,
            COUNT(*) as trades,
            AVG(exit_pnl_percent) as avg_pnl,
            SUM(CASE WHEN exit_pnl_percent >= 50 THEN 1 ELSE 0 END) * 100.0 / COUNT(*) as win_rate
        FROM positions
        WHERE status = 'closed' AND exit_time >= ? AND ai_narrative_tier IS NOT NULL
        GROUP BY ai_narrative_tier
        ORDER BY
            CASE ai_narrative_tier
                WHEN 'TIER_S' THEN 1
                WHEN 'TIER_A' THEN 2
                WHEN 'TIER_B' THEN 3
                WHEN 'TIER_C' THEN 4
                ELSE 5
            END
    `).all(cutoff);

    console.log(`\n${'═'.repeat(70)}`);
    console.log(`🎭 AI 叙事等級績效分析`);
    console.log(`${'═'.repeat(70)}\n`);

    if (byTier.length === 0) {
        console.log('無足夠數據（需要 ai_narrative_tier 字段）');
        return null;
    }

    for (const row of byTier) {
        const emoji = row.ai_narrative_tier === 'TIER_S' ? '🏆' :
                     row.ai_narrative_tier === 'TIER_A' ? '🥇' :
                     row.ai_narrative_tier === 'TIER_B' ? '🥈' : '🥉';

        console.log(`${emoji} ${row.ai_narrative_tier}: ${row.trades} 筆, 胜率 ${row.win_rate?.toFixed(1)}%, 平均 ${row.avg_pnl?.toFixed(1)}%`);
    }

    // 計算叙事等級與收益的相關性
    // 如果 TIER_S > TIER_A > TIER_B > TIER_C，說明叙事評分有效
    const tierOrder = ['TIER_S', 'TIER_A', 'TIER_B', 'TIER_C'];
    const orderedPnl = tierOrder.map(t => byTier.find(r => r.ai_narrative_tier === t)?.avg_pnl || 0);

    let monotonic = true;
    for (let i = 1; i < orderedPnl.length; i++) {
        if (orderedPnl[i] > orderedPnl[i-1]) monotonic = false;
    }

    console.log(`\n叙事等級與收益單調性: ${monotonic ? '✅ 符合預期' : '❌ 不符合預期'}`);

    return { byTier, monotonic };
}

function analyzeRejectedSignals(windowDays = 7) {
    const cutoff = new Date(Date.now() - windowDays * 24 * 60 * 60 * 1000).toISOString();

    const byStage = db.prepare(`
        SELECT
            rejection_stage,
            COUNT(*) as total,
            AVG(would_have_profit) as avg_avoided_pnl,
            SUM(CASE WHEN would_have_profit < -20 THEN 1 ELSE 0 END) as dodged_bullets,
            SUM(CASE WHEN would_have_profit > 50 THEN 1 ELSE 0 END) as missed_gains
        FROM rejected_signals
        WHERE created_at >= ? AND tracking_completed = 1
        GROUP BY rejection_stage
    `).all(cutoff);

    console.log(`\n${'═'.repeat(70)}`);
    console.log(`🛡️ 拒絕信號追蹤分析 (防守價值)`);
    console.log(`${'═'.repeat(70)}\n`);

    if (byStage.length === 0) {
        console.log('無追蹤數據');
        return null;
    }

    for (const row of byStage) {
        const emoji = row.avg_avoided_pnl < 0 ? '✅' : '❌';
        console.log(`${emoji} ${row.rejection_stage}`);
        console.log(`   拒絕: ${row.total} 筆 | 平均假設收益: ${row.avg_avoided_pnl?.toFixed(1)}%`);
        console.log(`   🛡️ 躲過大跌: ${row.dodged_bullets} | ❌ 錯過大漲: ${row.missed_gains}`);
    }

    return byStage;
}

function generateRecommendations(sourceAnalysis, narrativeAnalysis) {
    const recommendations = [];

    // 基於來源分析
    for (const source of sourceAnalysis.bySource) {
        if (source.trades < 10) continue; // 樣本不足

        if (source.avg_pnl < -10 && source.win_rate < 30) {
            recommendations.push({
                priority: 'HIGH',
                module: source.entry_source,
                action: 'DISABLE_OR_RESTRICT',
                reason: `胜率 ${source.win_rate?.toFixed(1)}%, 平均虧損 ${source.avg_pnl?.toFixed(1)}%`
            });
        } else if (source.avg_pnl > sourceAnalysis.baseline.avg_pnl * 1.5) {
            recommendations.push({
                priority: 'INFO',
                module: source.entry_source,
                action: 'INCREASE_WEIGHT',
                reason: `表現優於基準 50%+`
            });
        }
    }

    // 基於叙事分析
    if (narrativeAnalysis && !narrativeAnalysis.monotonic) {
        recommendations.push({
            priority: 'MEDIUM',
            module: 'ai_narrative',
            action: 'REVIEW_SCORING',
            reason: '叙事等級與收益不呈單調關係，評分邏輯可能無效'
        });
    }

    console.log(`\n${'═'.repeat(70)}`);
    console.log(`💡 優化建議`);
    console.log(`${'═'.repeat(70)}\n`);

    if (recommendations.length === 0) {
        console.log('目前無明確建議，請累積更多數據');
    } else {
        for (const rec of recommendations) {
            const emoji = rec.priority === 'HIGH' ? '🔴' : rec.priority === 'MEDIUM' ? '🟡' : '🟢';
            console.log(`${emoji} [${rec.priority}] ${rec.module}`);
            console.log(`   動作: ${rec.action}`);
            console.log(`   原因: ${rec.reason}\n`);
        }
    }

    return recommendations;
}

// 主函數
async function main() {
    const windowDays = parseInt(process.argv[2]) || 7;

    const sourceAnalysis = analyzeBySource(windowDays);
    const narrativeAnalysis = analyzeByNarrativeTier(windowDays);
    analyzeRejectedSignals(windowDays);
    generateRecommendations(sourceAnalysis, narrativeAnalysis);

    // 保存快照到 module_performance 表
    savePerformanceSnapshot(sourceAnalysis, windowDays);
}

function savePerformanceSnapshot(analysis, windowDays) {
    const snapshotDate = new Date().toISOString().split('T')[0];

    for (const source of analysis.bySource) {
        try {
            db.prepare(`
                INSERT OR REPLACE INTO module_performance (
                    snapshot_date, window_days, module_name,
                    total_trades, win_count, win_rate,
                    total_pnl, avg_pnl,
                    baseline_win_rate, baseline_avg_pnl, relative_performance,
                    sample_size_sufficient
                ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
            `).run(
                snapshotDate,
                windowDays,
                source.entry_source || 'unknown',
                source.trades,
                source.wins,
                source.win_rate,
                source.trades * source.avg_pnl,
                source.avg_pnl,
                analysis.baseline.win_rate,
                analysis.baseline.avg_pnl,
                analysis.baseline.avg_pnl ?
                    (source.avg_pnl - analysis.baseline.avg_pnl) / Math.abs(analysis.baseline.avg_pnl) * 100 : 0,
                source.trades >= 10 ? 1 : 0
            );
        } catch (e) {
            // 忽略
        }
    }
}

main().catch(console.error);
```

import Database from 'better-sqlite3';

async function trackRejectedSignals() {

        LIMIT 50
    `).all();

    console.log(`[TrackRejected] 需追蹤 ${toTrack.length} 筆信號`);

    for (const signal of toTrack) {

            const wouldHaveProfit = signal.price_at_rejection > 0 ?

            db.prepare(`

                    would_have_profit = ?,
                    tracking_completed = 1,
                    tracked_at = datetime('now')
                WHERE id = ?
            `).run(currentPrice, wouldHaveProfit, signal.id);

        } catch (e) {

#!/usr/bin/env node
/**
 * A/B 測試報告
 * 對比 control 組和 treatment 組的表現
 */

import Database from 'better-sqlite3';

const db = new Database('./data/sentiment_arb.db');

function generateABReport(windowDays = 14) {
    const cutoff = new Date(Date.now() - windowDays * 24 * 60 * 60 * 1000).toISOString();

    const groups = db.prepare(`
        SELECT
            experiment_group,
            COUNT(*) as trades,
            AVG(exit_pnl_percent) as avg_pnl,
            SUM(CASE WHEN exit_pnl_percent >= 50 THEN 1 ELSE 0 END) * 100.0 / COUNT(*) as win_rate,
            STDEV(exit_pnl_percent) as std_dev
        FROM positions
        WHERE status = 'closed'
        AND exit_time >= ?
        AND experiment_group IS NOT NULL
        GROUP BY experiment_group
    `).all(cutoff);

    console.log(`\n${'═'.repeat(70)}`);
    console.log(`🔬 A/B 測試報告 (過去 ${windowDays} 天)`);
    console.log(`${'═'.repeat(70)}\n`);

    if (groups.length < 2) {
        console.log('數據不足，需要 control 和 treatment 兩組數據');
        return;
    }

    const control = groups.find(g => g.experiment_group === 'control');
    const treatment = groups.find(g => g.experiment_group === 'treatment');

    console.log(`Control 組:    ${control?.trades || 0} 筆, 胜率 ${control?.win_rate?.toFixed(1) || 0}%, 平均 ${control?.avg_pnl?.toFixed(1) || 0}%`);
    console.log(`Treatment 組:  ${treatment?.trades || 0} 筆, 胜率 ${treatment?.win_rate?.toFixed(1) || 0}%, 平均 ${treatment?.avg_pnl?.toFixed(1) || 0}%`);

    if (control && treatment && control.trades >= 10 && treatment.trades >= 10) {
        const diff = treatment.avg_pnl - control.avg_pnl;
        const significant = Math.abs(diff) > (control.std_dev || 10); // 簡化的顯著性判斷

        console.log(`\n差異: ${diff > 0 ? '+' : ''}${diff.toFixed(1)}%`);
        console.log(`統計顯著: ${significant ? '✅ 是' : '❌ 否（需更多數據）'}`);

        if (significant && diff > 0) {
            console.log(`\n💡 建議: Treatment 組表現更好，考慮全量上線`);
        } else if (significant && diff < 0) {
            console.log(`\n💡 建議: Treatment 組表現更差，考慮回滾`);
        }
    }
}

/**
 * 模組考核與自動調權
 * 基於 module_performance 表的數據
 */
evaluateModules() {
    const config = {
        // 最低貢獻線
        minContribution: {
            ultra_fast_track: { minTrades: 10, minWinRate: 35, minAvgPnl: -5 },
            flash_scout: { minTrades: 5, minWinRate: 30, minAvgPnl: -10 },
            ai_narrative: { minCorrelation: 0.2 }, // 叙事分與收益的相關性
            waiting_room: { minTrades: 20, minWinRate: 30, minAvgPnl: -5 }
        },

        // 達標獎勵
        bonusThreshold: { winRate: 50, avgPnl: 20 },

        // 調整幅度
        penaltyStep: 0.2,  // 未達標降權20%
        bonusStep: 0.1     // 超標升權10%
    };

    const results = [];
    const recentPerf = this.db.prepare(`
        SELECT * FROM module_performance
        WHERE snapshot_date = (SELECT MAX(snapshot_date) FROM module_performance)
        AND window_days = 7
        AND sample_size_sufficient = 1
    `).all();

    for (const perf of recentPerf) {
        const moduleName = perf.module_name;
        const minReq = config.minContribution[moduleName];

        if (!minReq) continue;

        let action = 'MAINTAIN';
        let adjustment = 0;
        let reason = '';

        // 檢查是否達到最低貢獻線
        const meetsMin = (!minReq.minTrades || perf.total_trades >= minReq.minTrades) &&
                        (!minReq.minWinRate || perf.win_rate >= minReq.minWinRate) &&
                        (!minReq.minAvgPnl || perf.avg_pnl >= minReq.minAvgPnl);

        if (!meetsMin) {
            action = 'PENALIZE';
            adjustment = -config.penaltyStep;
            reason = `未達最低貢獻線: 交易${perf.total_trades}, 胜率${perf.win_rate?.toFixed(1)}%, 平均${perf.avg_pnl?.toFixed(1)}%`;
        } else if (perf.win_rate >= config.bonusThreshold.winRate ||
                   perf.avg_pnl >= config.bonusThreshold.avgPnl) {
            action = 'BONUS';
            adjustment = config.bonusStep;
            reason = `超額表現: 胜率${perf.win_rate?.toFixed(1)}%, 平均${perf.avg_pnl?.toFixed(1)}%`;
        }

        results.push({ moduleName, action, adjustment, reason, perf });

        // 執行調整
        if (adjustment !== 0) {
            this.adjustModuleWeight(moduleName, adjustment);
        }
    }

    return results;
}

/**
 * 調整模組權重
 */
adjustModuleWeight(moduleName, adjustment) {
    const currentWeight = this.dynamicWeights.sources[moduleName] || 1.0;
    const newWeight = Math.max(0.1, Math.min(2.0, currentWeight + adjustment));

    this.dynamicWeights.sources[moduleName] = newWeight;

    // 記錄調整
    this.recordAdjustment({
        type: 'module_evaluation',
        target: moduleName,
        oldValue: currentWeight,
        newValue: newWeight,
        reason: `自動考核調整`
    });

    console.log(`[ModuleEval] ${moduleName}: ${currentWeight.toFixed(2)} → ${newWeight.toFixed(2)}`);

    // 如果權重降到 0.3 以下，發出警告
    if (newWeight <= 0.3) {
        console.log(`[ModuleEval] ⚠️ ${moduleName} 權重過低，建議檢查或關閉`);
        this.emit('module-critical', { moduleName, weight: newWeight });
    }
}

/**
 * 評估 AI 叙事模組的有效性
 * 基於叙事等級與收益的相關性
 */
evaluateNarrativeEffectiveness() {
    const data = this.db.prepare(`
        SELECT
            ai_narrative_tier,
            AVG(exit_pnl_percent) as avg_pnl,
            COUNT(*) as trades
        FROM positions
        WHERE status = 'closed'
        AND ai_narrative_tier IS NOT NULL
        AND exit_time >= datetime('now', '-7 days')
        GROUP BY ai_narrative_tier
        HAVING COUNT(*) >= 5
    `).all();

    if (data.length < 2) {
        return { effective: null, reason: '數據不足' };
    }

    // 檢查單調性：TIER_S > TIER_A > TIER_B > TIER_C
    const tierOrder = ['TIER_S', 'TIER_A', 'TIER_B', 'TIER_C'];
    const orderedPnl = tierOrder
        .map(t => data.find(d => d.ai_narrative_tier === t))
        .filter(Boolean)
        .map(d => d.avg_pnl);

    let monotonic = true;
    for (let i = 1; i < orderedPnl.length; i++) {
        if (orderedPnl[i] >= orderedPnl[i-1]) {
            monotonic = false;
            break;
        }
    }

    // 計算簡化的相關係數
    const tierScore = { 'TIER_S': 4, 'TIER_A': 3, 'TIER_B': 2, 'TIER_C': 1 };
    const points = data.map(d => ({ x: tierScore[d.ai_narrative_tier], y: d.avg_pnl }));
    const correlation = this.calculateCorrelation(points);

    const effective = monotonic && correlation > 0.3;

    return {
        effective,
        monotonic,
        correlation: correlation.toFixed(2),
        data,
        recommendation: effective ?
            '叙事評分有效，保持使用' :
            '叙事評分效果不明顯，考慮簡化或關閉'
    };
}

/**
 * 簡單相關係數計算
 */
calculateCorrelation(points) {
    if (points.length < 2) return 0;

    const n = points.length;
    const sumX = points.reduce((a, p) => a + p.x, 0);
    const sumY = points.reduce((a, p) => a + p.y, 0);
    const sumXY = points.reduce((a, p) => a + p.x * p.y, 0);
    const sumX2 = points.reduce((a, p) => a + p.x * p.x, 0);
    const sumY2 = points.reduce((a, p) => a + p.y * p.y, 0);

    const numerator = n * sumXY - sumX * sumY;
    const denominator = Math.sqrt((n * sumX2 - sumX * sumX) * (n * sumY2 - sumY * sumY));

    return denominator === 0 ? 0 : numerator / denominator;
}
